#include <windows.h>
#include <GL/glut.h>

int windowWidth = 640;
int windowHeight = 480;
int windowPosX = 50;
int windowPosY = 50;

float zcube = -1;
float zpyramid = -1;
float zspeed = 1;
int refreshMills = 1000/30;
bool paused = false, fullScreenMode = true;
float zSpeedSaved;
void background() {
    glClearColor(0, 0, 0, 1);
    glEnable(GL_DEPTH_TEST);
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
    glLoadIdentity();
    
    ////////////////Cube/////////////////
    glTranslatef(1.5, 0, zcube);

    glBegin(GL_QUADS);
    // Top face (y = 1)
    glColor3f(0, 1, 0); // Green
    glVertex3f(1, 1, -1);
    glVertex3f(-1, 1, -1);
    glVertex3f(-1, 1, 1);
    glVertex3f(1, 1, 1);

    // Bottom face (y = -1)
    glColor3f(1, 0.5, 0); // Orange
    glVertex3f(1, -1, 1);
    glVertex3f(-1, -1, 1);
    glVertex3f(-1, -1, -1);
    glVertex3f(1, -1, -1);

    // Front face  (z = 1)
    glColor3f(1, 0, 0); // Red
    glVertex3f(1, 1, 1);
    glVertex3f(-1, 1, 1);
    glVertex3f(-1, -1, 1);
    glVertex3f(1, -1, 1);

    // Back face (z = -1)
    glColor3f(1, 1, 0); // Yellow
    glVertex3f(1, -1, -1);
    glVertex3f(-1, -1, -1);
    glVertex3f(-1, 1, -1);
    glVertex3f(1, 1, -1);

    // Right face (x = 1)
    glColor3f(1, 0, 1); // Magenta
    glVertex3f(1, 1, -1);
    glVertex3f(1, 1, 1);
    glVertex3f(1, -1, 1);
    glVertex3f(1, -1, -1);

    // Left face (x = -1)
    glColor3f(0, 0, 1); // Blue
    glVertex3f(-1, 1, 1);
    glVertex3f(-1, 1, -1);
    glVertex3f(-1, -1, -1);
    glVertex3f(-1, -1, 1);

    glEnd();

    ////////////////Pyramid/////////////////
    glLoadIdentity();
    glTranslatef(-1.5, 0, zpyramid);

    glBegin(GL_TRIANGLES);
    //front
    glColor3f(1, 0, 0); // Red
    glVertex3f(0, 1, 0);
    glColor3f(0, 1, 0); // Green
    glVertex3f(-1, -1, 1);
    glColor3f(0, 0, 1); // Blue
    glVertex3f(1, -1, 1);

    //back
    glColor3f(1, 0, 0); // Red
    glVertex3f(0, 1, 0);
    glColor3f(0, 1, 0); // Green
    glVertex3f(1, -1, -1);
    glColor3f(0, 0, 1); // Blue
    glVertex3f(-1, -1, -1);

    //right
    glColor3f(1, 0, 0); // Red
    glVertex3f(0, 1, 0);
    glColor3f(0, 0, 1); // Blue
    glVertex3f(1, -1, 1);
    glColor3f(0, 1, 0); // Green
    glVertex3f(1, -1, -1);

    //left
    glColor3f(1, 0, 0);   // Red
    glVertex3f(0, 1, 0);
    glColor3f(0, 0, 1);   // Blue
    glVertex3f(-1, -1, -1);
    glColor3f(0, 1, 0);   // Green
    glVertex3f(-1, -1, 1);
    glEnd();

    glutSwapBuffers();
}

void timer(int value) {
    glutPostRedisplay();
    glutTimerFunc(refreshMills, timer, 0);

    zcube += zspeed;
    zpyramid += zspeed;
    if (zcube > -1) {
        zcube = -1;
        zspeed = -zspeed;
    }
    else if (zcube < -99) {
        zcube = -99;
        zspeed = -zspeed;
    }
}

void reshape(GLsizei width, GLsizei height) {
    if (height == 0) height = 1;
    float aspect = (float)width / height;
    glViewport(0, 0, width, height);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluPerspective(45, aspect, 0.1, 100);
    glMatrixMode(GL_MODELVIEW);
}

void mouse(int button, int state, int x, int y) {
    if (button == GLUT_LEFT_BUTTON && state == GLUT_UP) {
        paused = !paused;
        if (paused) {
            zSpeedSaved = zspeed;
            zspeed = 0;
        }
        else {
            zspeed = zSpeedSaved;
        }
    }
}


void specialKeys(int key, int x, int y) {
    switch (key) {
    case GLUT_KEY_F1:
        fullScreenMode = !fullScreenMode;
        if (fullScreenMode)
        {
            //save parameters for restoring later
            windowPosX = glutGet(GLUT_WINDOW_X);
            windowPosY = glutGet(GLUT_WINDOW_Y);
            windowWidth = glutGet(GLUT_WINDOW_WIDTH);
            windowHeight = glutGet(GLUT_WINDOW_HEIGHT);
            glutFullScreen();
        }
        else
        {
            glutReshapeWindow(windowWidth, windowHeight);
            glutPositionWindow(windowPosX, windowPosY);
        }
        break;
    case GLUT_KEY_RIGHT:
        zspeed *= 1.05f;
        break;
    case GLUT_KEY_LEFT:
        zspeed *= 0.95f;
        break;
    }
}

void keyboard(unsigned char key, int x, int y) {
    switch (key) {
    case 27:
        exit(0);
        break;

    }
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE);
    glutInitWindowSize(640, 480);
    glutInitWindowPosition(50, 50);
    glutCreateWindow("3D animation");
    glutDisplayFunc(display);
    glutReshapeFunc(reshape);
    background();
    glutTimerFunc(0, timer, 0);
    glutMouseFunc(mouse);
    glutSpecialFunc(specialKeys);
    glutKeyboardFunc(keyboard);
    glutMainLoop();
    return 0;
}
