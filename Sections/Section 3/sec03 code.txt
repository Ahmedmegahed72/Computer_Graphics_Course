#include<Windows.h> // get screen size
#include<GL\glut.h>

float xp, deltax = 0.3, sWidth, sHeight;

void draw();
void reshape(int, int);
void backGround();
void timer(int);

int main(int argc, char** argv) {
	// glut init
	glutInit(&argc, argv);
	glutInitDisplayMode(GLUT_RGB | GLUT_DOUBLE);

	// create window
	float width = GetSystemMetrics(SM_CXSCREEN);
	float height = GetSystemMetrics(SM_CYSCREEN);
	glutInitWindowSize(width, height);
	glutInitWindowPosition(0,0);
	glutCreateWindow("second Section");

	// call back functions
	glutDisplayFunc(draw);
	glutReshapeFunc(reshape);
	glutTimerFunc(0, timer, 0);
	backGround();

	// main loop
	glutMainLoop();
}
void draw() {
	glClear(GL_COLOR_BUFFER_BIT);
	glLoadIdentity();

	glBegin(GL_QUADS);
	glColor3f(1, 1, 0);
	glVertex2f(xp, 0);
	glVertex2f(xp + 2, 0);
	glVertex2f(xp + 2, 2);
	glVertex2f(xp, 2);
	glEnd();
	
	glutSwapBuffers();
}

void reshape(int w, int h) {
	float ratio;
	if (h == 0)
		h = 1;
	ratio = (float)w / h;
	glViewport(0, 0, w, h);

	glMatrixMode(GL_PROJECTION);
	glLoadIdentity();
	if (w >= h) 
	{
		gluOrtho2D(-10 * ratio, 10 * ratio, -10, 10);
		sWidth = 10 * ratio;
		sHeight = 10;
	}
	else 
	{
		gluOrtho2D(-10, 10, -10 / ratio, 10 / ratio);
		sWidth = 10;
		sHeight = 10 / ratio;
	}

	glMatrixMode(GL_MODELVIEW);
}

void backGround() {
	// color [0,1]
	glClearColor(.5, .4, 1, 0);
}

void timer(int v) {
	// repaint
	glutPostRedisplay();
	glutTimerFunc(1000 / 25, timer, 0);

	xp += deltax;
	if (xp > sWidth - 2 || xp < -sWidth)
		deltax = -deltax;
}
